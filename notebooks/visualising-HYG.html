Visualising the HYG Database

Introduction

Here Ill be exploring the HYG Database.

# loading libraries
library(tidyverse)
library(GGally)
library(ggthemes)

# sourcing useful functions
source("../utils/render_table.R", local = TRUE)

Load the data

hyg_data <- read_csv("../data/hygdata_v3.csv")

View the data

hyg_data %>% 
  head(20) %>% 
  render_table()
hyg_data %>% 
  select(ra:absmag) %>% 
  sample_n(1000) %>% 
  ggpairs(aes(colour = 'steelblue', alpha = 0.01),
          lower = list(continuous = wrap("smooth", colour = "steelblue")),
          diag = list(continuous = wrap("barDiag", fill = "steelblue"))) +
  theme_few()

hyg_data %>% 
  select(x:vz) %>% 
  sample_n(10000) %>% 
  ggpairs(aes(colour = 'steelblue', alpha = 1),
          lower = list(continuous = wrap("smooth", colour = "steelblue")),
          diag = list(continuous = wrap("barDiag", fill = "steelblue"))) +
  theme_few()
